# isIp

▸ isIp(value: string, type?: IpTypes): boolean

Defined in is/ip.ts:26

name isIp

summary Tests if the value is a valid IP address

description Checks to see if the value is a valid IP address. Optionally check for either v4/v6

example 

import { isIp } from '@polkadot/util';

isIp('192.168.0.1')); // => true
isIp('1:2:3:4:5:6:7:8'); // => true
isIp('192.168.0.1', 'v6')); // => false
isIp('1:2:3:4:5:6:7:8', 'v4'); // => false
